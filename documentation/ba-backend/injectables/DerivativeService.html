    async calculateDerivative(name: string, namespace: string, lastEvent: number, firstEvent: number, metricType: string) {
        let result;
        let negativeGradient = 0;
        let positiveGradient = 0;
        let positivePercentage;
        result = await lastValueFrom(this.getMetrics(name, namespace, lastEvent, firstEvent, metricType))
        if (Array.isArray(result.data.data.result) && result.data.data.result.length) {
            let metrics: Array<any> = result.data.data.result.pop().values;
            for (let i = 0; i < metrics.length - 1; i++) {
                if ((parseFloat(metrics[i][1]) - parseFloat(metrics[i + 1][1])) < 0) {
                    negativeGradient = negativeGradient + 1;
                } else {
                    positiveGradient = positiveGradient + 1;
                }
            }
            positivePercentage = positiveGradient / metrics.length
            return positivePercentage;
        }

        return positivePercentage = -1
    }
